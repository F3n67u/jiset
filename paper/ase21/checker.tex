\begin{itemize}

\end{itemize}

\subsection{Parameter Arity Checker}
\begin{itemize}

\end{itemize}

\subsection{Type Checker}
\begin{itemize}

\end{itemize}

\subsection{Assertion Checker}
\begin{itemize}
  \item `typeof` assertions
  \item `instance-of` assertions
  \item reachability assertions
\end{itemize}
